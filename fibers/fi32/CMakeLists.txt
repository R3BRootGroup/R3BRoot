# Create a library called "libR3BFi32" which includes the source files
# given in the array. The extension is already found. Any number of 
# sources could be listed here.

Set(SYSTEM_INCLUDE_DIRECTORIES 
${SYSTEM_INCLUDE_DIRECTORIES}
${BASE_INCLUDE_DIRECTORIES}
)

set(INCLUDE_DIRECTORIES
${R3BROOT_SOURCE_DIR}/fibers/fi32
${R3BROOT_SOURCE_DIR}/fiber
${R3BROOT_SOURCE_DIR}/r3bbase
${R3BROOT_SOURCE_DIR}/r3bdata
${R3BROOT_SOURCE_DIR}/r3bdata/fibData
${R3BROOT_SOURCE_DIR}/tcal
${R3BROOT_SOURCE_DIR}/tracking
)

include_directories(${INCLUDE_DIRECTORIES})
include_directories(SYSTEM ${SYSTEM_INCLUDE_DIRECTORIES})

set(LINK_DIRECTORIES
${ROOT_LIBRARY_DIR}
${FAIRROOT_LIBRARY_DIR}
) 

link_directories(${LINK_DIRECTORIES})

set(SRCS
R3BFi32.cxx  
R3BFi32Cal2Hit.cxx
R3BFi32Mapped2Cal.cxx
R3BFi32ContFact.cxx
R3BFi32Digitizer.cxx
R3BFi32DigitizerCal.cxx
)

CHANGE_FILE_EXTENSION(*.cxx *.h HEADERS "${SRCS}")

Set(LINKDEF Fi32LinkDef.h)
Set(LIBRARY_NAME R3BFi32)
Set(DEPENDENCIES
    R3BBunchedFiber R3BTracking R3BData)

GENERATE_LIBRARY()
